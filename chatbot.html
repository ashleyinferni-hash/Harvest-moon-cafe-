class CafeAssistant: """ A simple AI-like assistant to match cafe items based on tags. """ def __init__(self): # A dictionary representing the cafe's menu with associated tags self.menu = { "Espresso": ["drink", "hot", "caffeine"], "Decaf Latte": ["drink", "hot", "milk"], "Blueberry Muffin": ["food", "sweet", "pastry"], "Vegan Sandwich": ["food", "savory", "vegan", "lunch"], "Iced Tea": ["drink", "cold", "refreshing"], "Gluten-Free Brownie": ["food", "sweet", "gluten-free"] } def recommend_item(self, preference_tag: str) -> list: """ Finds menu items that match the provided tag. Args: preference_tag: A string (e.g., 'vegan', 'hot', 'sweet'). Returns: A list of menu item names that match the tag. """ matches = [] # Convert tag to lowercase for case-insensitive matching target_tag = preference_tag.lower() for item, tags in self.menu.items(): # Check if the target tag is in the item's list of tags if target_tag in tags: matches.append(item) return matches # --- Example Usage --- # 1. Initialize the assistant assistant = CafeAssistant() # 2.
